-- TABLA "machos"
CREATE TABLE machos (

macho_numero INT,
macho_arete VARCHAR(50),
macho_tatuaje VARCHAR(50),
macho_raza VARCHAR(50),
madre_numero INT,
madre_arete VARCHAR(50),
madre_tatuaje VARCHAR(50),
madre_raza VARCHAR(50),
padre_numero INT,
padre_arete VARCHAR(50),
padre_tatuaje VARCHAR(50),
padre_raza VARCHAR(50),
macho_color VARCHAR(50),
macho_talla  VARCHAR(50),
macho_pelo VARCHAR(50),
macho_condicion VARCHAR(50),
macho_estatus VARCHAR(50),
macho_potrero VARCHAR(50),
macho_lote VARCHAR(50),
macho_estado_re VARCHAR(50),
macho_finado VARCHAR(50),
macho_edad_actual VARCHAR(50),
macho_edad_destete VARCHAR(50),
macho_edad_venta VARCHAR(50),
macho_peso_nacimiento VARCHAR(50),
macho_peso_actual VARCHAR(50),
macho_peso_destete VARCHAR(50),
macho_peso_venta VARCHAR(50),
macho_gan_peso_dia VARCHAR(50),
macho_gan_peso_mes VARCHAR(50),
macho_peso_3meses VARCHAR(50),
macho_fecha_nacimiento VARCHAR(20),
macho_fecha_destete VARCHAR(20),
macho_fecha_aretado VARCHAR(20),
macho_fecha_tatuaje VARCHAR(20),
macho_fecha_fierro VARCHAR(20),
macho_fecha_venta VARCHAR(20),
macho_foto VARCHAR(150),
macho_foto_fierro  VARCHAR(150),
macho_observaciones VARCHAR(200),
ultima_ejecucion DATETIME NULL DEFAULT NULL,
fecha_registro DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
