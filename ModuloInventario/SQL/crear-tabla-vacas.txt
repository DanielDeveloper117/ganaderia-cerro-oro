CREATE TABLE vacas (

vaca_numero INT,
vaca_arete VARCHAR(50),
vaca_tatuaje VARCHAR(50),
vaca_raza VARCHAR(50),
madre_numero INT,
madre_arete VARCHAR(50),
madre_tatuaje VARCHAR(50),
madre_raza VARCHAR(50),
padre_numero INT,
padre_arete VARCHAR(50),
padre_tatuaje VARCHAR(50),
padre_raza VARCHAR(50),
vaca_color VARCHAR(50),
vaca_talla  VARCHAR(50),
vaca_pelo VARCHAR(50),
vaca_condicion VARCHAR(50),
vaca_estatus VARCHAR(50),
vaca_potrero VARCHAR(50),
vaca_lote VARCHAR(50),
vaca_estado_re VARCHAR(50),
vaca_celo VARCHAR(50),
vaca_partos INT,
vaca_estado_pal VARCHAR(50),
vaca_finada VARCHAR(50),
vaca_edad_actual VARCHAR(50),
vaca_edad_destete VARCHAR(50),
vaca_edad_venta VARCHAR(50),
vaca_peso_nacimiento VARCHAR(50),
vaca_peso_actual VARCHAR(50),
vaca_peso_destete VARCHAR(50),
vaca_peso_venta VARCHAR(50),
vaca_gan_peso_dia VARCHAR(50),
vaca_gan_peso_mes VARCHAR(50),

vaca_fecha_nacimiento VARCHAR(20),
vaca_fecha_destete VARCHAR(20),
vaca_fecha_aretado VARCHAR(20),
vaca_fecha_tatuaje VARCHAR(20),
vaca_fecha_fierro VARCHAR(20),
vaca_fecha_probable VARCHAR(20),
vaca_fecha_venta VARCHAR(20),
vaca_leche_dia DECIMAL(6, 3),
vaca_leche_mes DECIMAL(8, 3),
vaca_leche_comentario  VARCHAR(100),
vaca_foto VARCHAR(150),
vaca_foto_fierro  VARCHAR(150),
vaca_observaciones VARCHAR(200),
ultima_ejecucion DATETIME NULL DEFAULT NULL,
fecha_registro DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
