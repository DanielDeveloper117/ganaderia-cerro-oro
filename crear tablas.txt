    fecha_nacimiento VARCHAR(20),

    fecha_venta VARCHAR(20),
    edad_actual VARCHAR(20),
    edad_destete VARCHAR(30),
    edad_venta VARCHAR(30),
    peso_nacimiento VARCHAR(30),
    peso_3meses VARCHAR(30),
    peso_destete VARCHAR(30),
    peso_venta VARCHAR(30),
    ganancia_peso_dia VARCHAR(30),
    ganancia_peso_mes VARCHAR(30),
    finado VARCHAR(20),
    cria_num VARCHAR(30),
    cria_arete VARCHAR(30),
    destino_cria VARCHAR(30),
    vaca_num INT,
    vaca_arete VARCHAR(30),
    vaca_raza VARCHAR(50),
    fecha_aretado VARCHAR(30),
    estatus VARCHAR(30),
    potrero VARCHAR(50),
    lote VARCHAR(30),
    estado_reproductivo VARCHAR(30),
    celo VARCHAR(50),
    parto_num VARCHAR(30),
    estado_palpacion VARCHAR(50),
    fecha_probable VARCHAR(30),
    foto VARCHAR(100),

    fecha_registro DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
);
